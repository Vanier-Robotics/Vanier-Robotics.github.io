107 void update()
108 {
109 // During each update, check which button are pressed and update necessary callbacks
110
111 // Button bindings
112 for (int i = 0; i < m_buttonBindings.getSize(); i++)
113 {
114 bool status = Crc::CrcLib::ReadDigitalChannel(m_buttonBindings[i].buttonID);
115 if (status != m_buttonBindings[i].lastValue)
116 {
117 m_buttonBindings[i].callback(status);
118 }
119
120 m_buttonBindings[i].lastValue = status;
121 }
122
123 // Toggle button bindings
124 for (int i = 0; i < m_toggleButtonBindings.getSize(); i++)
125 {
126 bool status = Crc::CrcLib::ReadDigitalChannel(m_toggleButtonBindings[i].buttonID);
127 if (status && !m_toggleButtonBindings[i].lastValue) // the button was just pressed
128 {
129 m_toggleButtonBindings[i].isToggled != m_toggleButtonBindings[i].isToggled;
130 m_toggleButtonBindings[i].callback(m_toggleButtonBindings[i].isToggled);
131 }
132
133 m_toggleButtonBindings[i].lastValue = status;
134 }
135
136 // Digital Sensor Bindings verification
137 for (int i = 0; i < m_digitalSensorBindings.getSize(); i++)
138 {
139 bool status = m_digitalSensorBindings[i].handle.getValue();
140 if (status != m_buttonBindings[i].lastValue)
141 {
142 m_digitalSensorBindings[i].callback(status);
