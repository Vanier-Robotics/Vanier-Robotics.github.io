158 if (m_toggleBindings[i].lastValue != status && status)
159 {
160 m_toggleBindings[i].isToggled != m_toggleBindings[i].isToggled;
161 m_toggleBindings[i].callback(m_toggleBindings[i].isToggled);
162 }

165 }
